add_namespace = NFEvents

country_event = {
	id = NFEvents.1
	title = "[SOV.GetName] Demands [MON.GetName]!"
	desc = "Today [SOV.GetName] sent an envoy to us with a proposition of an union. We are alone and in this world, and a union with [SOV.GetName might prove to be fruitful. Our people would be safe with the mighty army of [SOV.GetName] and we could possibly flourish with their established economy. Or we could refuse the union which would surely lead to war, but maybe we can hold them off!"
	picture = GFX_report_event_hitler_parade

	is_triggered_only = yes

	option = {
		name = "We accept the Union"
		ai_chance = {
			base = 30
			modifier = {
				factor = 0
				MON = { is_subject = yes }
			}
			modifier = {
				factor = 0
				NOT = { tag = MON }
				SOV = { on_same_continent = no }
			}
			modifier = {
				add = -15
				SOV = { has_army_size = { size < 40 } }
			}
			modifier = {
				add = 45
				SOV = { has_army_size = { size > 39 } }
			}
		}
		FROM = {
			country_event = { hours = 2 id = NFEvents.2 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}

	option = {
		name = "We Refuse!"
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				SOV = { has_army_size = { size > 39 } }
			}
			modifier = {
				add = 20
				SOV = { has_army_size = { size < 30 } }
			}
		}
		FROM = {
			country_event = { hours = 2 id = NFEvents.3 }
			if = {
				limit = { is_in_faction_with = MON }
				remove_from_faction = MON
			}
		}
	}
}

country_event = {
	id = NFEvents.3
	title = "[FROM.GetName] Refuses!"
	desc = "[FROM.GetName] Refused our proposed union! This is an insult to us that cannot go unanswered!"
	picture = GFX_report_event_german_troops

	is_triggered_only = yes

	option = {
		name = "It's time for war"
		create_wargoal = {
			type = annex_everything
			target = FROM
		}
		news_event = { id = static_news.3 hours = 6 }
	}
}

country_event = {
	id = NFEvents.2
	title = "[FROM.GetName] accepts!"
	desc = "[FROM.GetName] accepted our proposed union, their added strength will push us to greatness!"
	picture = GFX_report_event_german_speech

	is_triggered_only = yes

	option = {
		name = "A stronger Union!"
		every_country = {
			limit = { is_subject_of = MON }
			country_event = NFEvents.1
		}
		FROM = {
			every_owned_state = {
				add_core_of = SOV
			}
		}
		if = {
			limit = { FROM = { tag = MON } }
			annex_country = { target = MON transfer_troops = yes }
			add_political_power = 50
			add_named_threat = { threat = 2 name = "Russia annexed Mongolia" }
			news_event = { id = static_news.2 hours = 6 }
			set_country_flag = MON_annexed
		}
		else = {
			puppet = FROM
		}

	}
}

country_event = {
	id = NFEvents.28
	title = "Alliance Offer"
	desc = "We have been invited to an alliance with [SOV.GetName]. Should we accept the invitation?"
	picture = news_event_generic_sign_treaty1

	is_triggered_only = yes

	option = {
		name = "Yes"
		ai_chance = {
			factor = 3
			modifier = {
				factor = 0
				NOT = { has_government = SOV }
			}
			modifier = {
				factor = 0.5
				has_opinion = {
					target = SOV
					value < 100
				}
			}
			modifier = {
				factor = 0.5
				has_opinion = {
					target = SOV
					value < 50
				}
			}
			modifier = {
				factor = 0
				has_opinion = {
					target = SOV
					value < 0
				}
			}
		}

		if = {
			limit = {
				SOV = {
					is_faction_leader = no
				}
			}
			SOV = {
				leave_faction = yes
				create_faction = "nationalist_league"
			}
		}
		SOV = {
			add_to_faction = ROOT
		}
		hidden_effect = {
			news_event = { id = news.502 }
		}
	}

	option = {
		name = "No"
		ai_chance = { factor = 1 }
		hidden_effect = {
			news_event = { id = news.503 }
		}
	}
}

	is_triggered_only = yes

	option = {
		name = "We Accept"
		ai_chance = {
			base = 30
			modifier = {
				add = -15
				W10 = { has_army_size = { size < 40 } }
			}
			modifier = {
				add = 45
				W10 = { has_army_size = { size > 39 } }
			}
		}
		W10 = {
			country_event = { hours = 2 id = NFEvents.30 }
		}
	}

	option = {
		name = "We Refuse!"
		ai_chance  = {
			base = 10
			modifier = {
				factor = 0
				W10 = { has_army_size = { size > 39 } }
			}
			modifier = {
				add = 20
				W10 = { has_army_size = { size < 30 } }
			}
		}
		W10 = {
			country_event = { hours = 2 id = NFEvents.31 }
			if = {
				limit = { is_in_faction_with = TIB }

			}
		}
	}
